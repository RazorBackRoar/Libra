## **PYTHON IMPLEMENTATION TIPS**

**For PySide6 (standalone script):**
```python
import sys
from PySide6.QtWidgets import QApplication, QMainWindow, QWidget, QFrame, QHBoxLayout
from PySide6.QtCore import Qt

# Window setup
app = QApplication(sys.argv)
window = QMainWindow()
window.setGeometry(100, 100, 1300, 800)
window.setStyleSheet("background-color: #1a1a1a;")

# Create central widget and main layout
central_widget = QWidget()
main_layout = QHBoxLayout()
main_layout.setSpacing(0)
main_layout.setContentsMargins(0, 0, 0, 0)

# Left frame (Controls)
left_frame = QFrame()
left_frame.setFixedWidth(220)
left_frame.setStyleSheet("background-color: #2d2d2d;")

# Center frame (Table)
center_frame = QFrame()
center_frame.setStyleSheet("background-color: #1a1a1a;")

# Right frame (Inspector)
right_frame = QFrame()
right_frame.setFixedWidth(300)
right_frame.setStyleSheet("background-color: #2d2d2d;")

# Add frames to layout
main_layout.addWidget(left_frame)
main_layout.addWidget(center_frame, 1)  # stretch factor of 1 = expand
main_layout.addWidget(right_frame)

# Set layout and show
central_widget.setLayout(main_layout)
window.setCentralWidget(central_widget)
window.show()

sys.exit(app.exec())
```

**For PySide6 (inside a class - cleaner approach):**
```python
class VideoManagerApp(QMainWindow):
    def __init__(self):
        super().__init__()
        self.setGeometry(100, 100, 1300, 800)
        self.setStyleSheet("background-color: #1a1a1a;")
        
        # Create central widget and main layout
        central_widget = QWidget()
        main_layout = QHBoxLayout()
        main_layout.setSpacing(0)
        main_layout.setContentsMargins(0, 0, 0, 0)
        
        # Create frames
        left_frame = self.create_left_frame()
        center_frame = self.create_center_frame()
        right_frame = self.create_right_frame()
        
        # Add to layout
        main_layout.addWidget(left_frame)
        main_layout.addWidget(center_frame, 1)
        main_layout.addWidget(right_frame)
        
        central_widget.setLayout(main_layout)
        self.setCentralWidget(central_widget)
```
